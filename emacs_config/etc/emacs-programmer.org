* quickrun.el
#+begin_src emacs-lisp
  (use-package quickrun
    :after prog-mode
    :config
    ;; set python3 as default
    (quickrun-add-command "python" 
      '((:command . "python3") 
        (:exec . "%c %s") 
        (:tempfile . nil)) 
      :default "python"))

  ;; Set up Keybindings
    (dw/leader-key-def
    "r"  '(:ignore t :which-key "quickrun")
    "rr" 'quickrun
    "ra" 'quickrun-with-arg
    "rs" 'quickrun-shell
    "rc" 'quickrun-compile-only
    "re" 'quickrun-region)
#+end_src
